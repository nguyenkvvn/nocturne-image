# basic superbird uboot stuff
EnableSelinux=enforcing
Irq_check_en=0
active_slot=_a
avb2=0
bcb_cmd=get_avb_mode;get_valid_slot;
board_revision=4
boot_part=boot
jtag=disable

# board configuration
baudrate=115200
stderr=serial
stdin=serial
stdout=serial

# display setup
display_bpp=16
display_color_bg=0
display_color_fg=0xffff
display_color_index=16
display_height=800
display_init=1
display_layer=osd0
display_stack=unknown
display_width=480
fb_addr=0x1f800000
fb_height=800
fb_width=480
frac_rate_policy=1
osd_reverse=0
outputmode=panel
panel_type=lcd_8
video_reverse=0

# dtb config
dtb_mem_addr=0x1000000
fdt_high=0x20000000

# other stuff
sdcburncfg=aml_sdc_burn.ini
sdc_burning=sdc_burn ${sdcburncfg}
reboot_mode=normal
reboot_mode_android=normal
recovery_offset=0
recovery_part=recovery
wipe_cache=successful
wipe_data=successful
system_mode=1

# general configuration
loadaddr=1080000
lock=10001000
silent=off
upgrade_step=0

# always have firstboot after flash
firstboot=1

# boot checks
check_button=if gpio input GPIOA_3; then run do_usb_burning; elif gpio input GPIOAO_3; then run start_shell; fi;
check_charger=mw 0xFF6346DC 0x33000000;mw.b 0x1337DEAD 0x00 1;mw.b 0x1330DEAD 0x12 1;mw.b 0x1331DEAD 0x13 1;mw.b 0x1332DEAD 0x15 1;mw.b 0x1333DEAD 0x16 1;i2c dev 2;i2c read 0x35 0x3 1 0x1337DEAD;if cmp.b 0x1337DEAD 0x1330DEAD 1; then run storeboot;elif cmp.b 0x1337DEAD 0x1331DEAD 1; then run storeboot;elif cmp.b 0x1337DEAD 0x1332DEAD 1; then run storeboot;elif cmp.b 0x1337DEAD 0x1333DEAD 1; then run storeboot;else osd open;osd clear;imgread pic logo bad_charger $loadaddr;bmp display $bad_charger_offset;bmp scale;vout output ${outputmode};while true; do sleep 1; if gpio input GPIOAO_3; then run splash_boot; fi; i2c read 0x35 0x3 1 0x1337DEAD;if cmp.b 0x1337DEAD 0x1330DEAD 1; then run splash_boot;elif cmp.b 0x1337DEAD 0x1331DEAD 1; then run splash_boot;elif cmp.b 0x1337DEAD 0x1332DEAD 1; then run splash_boot;elif cmp.b 0x1337DEAD 0x1333DEAD 1; then run splash_boot;fi;i2c mw 0x35 0x09 0x8F 1;done;fi;
check_temp=boot_cooling

# boot functions
update=run usb_burning;
init_display=if test ${display_init} = 1; then osd open;osd clear;imgread pic logo bootup_spotify $loadaddr;bmp display $bootup_spotify_offset;bmp scale;vout output ${outputmode};fi;
start_shell=osd open;osd clear;imgread pic logo shell_mode $loadaddr;bmp display $shell_mode_offset;bmp scale;vout output ${outputmode}; update 1; setenv bootcmd "amlmmc env";
splash_boot=imgread pic logo bootup_spotify $loadaddr;bmp display $bootup_spotify_offset;bmp scale;run storeboot;
do_usb_burning=osd open;osd clear;imgread pic logo burn_mode $loadaddr;bmp display $burn_mode_offset;bmp scale;vout output ${outputmode}; update 1000;
usb_burning=update 1000
try_auto_burn=update 700 750;
sys_part=mmcblk0p12
kernel_part=10
get_boot_slot=if test $active_slot = _a; then setenv sys_part mmcblk0p12; elif test $active_slot = _b; then setenv sys_part mmcblk0p13; fi
get_kernel_slot=if test $active_slot = _a; then setenv kernel_part 10; elif test $active_slot = _b; then setenv kernel_part 11; fi
initargs=init=/sbin/init ramoops.pstore_en=1 ramoops.record_size=0x8000 ramoops.console_size=0x4000 rootfstype=ext4 console=ttyS0,115200n8 no_console_suspend earlycon=aml-uart,0xff803000

# boot
preboot=run bcb_cmd; run init_display; run storeargs; bcb uboot-command;
bootcmd=run storeboot;

storeargs=setenv bootargs ${initargs} root=/dev/${sys_part} ro rootwait skip_initramfs reboot_mode_android=${reboot_mode_android} logo=${display_layer},loaded,${fb_addr} fb_width=${fb_width} fb_height=${fb_height} vout=${outputmode},enable panel_type=${panel_type} frac_rate_policy=${frac_rate_policy} osd_reverse=${osd_reverse} video_reverse=${video_reverse} irq_check_en=${Irq_check_en} androidboot.selinux=${EnableSelinux} androidboot.firstboot=${firstboot} jtag=${jtag} uboot_version=${gitver};setenv bootargs ${bootargs} androidboot.hardware=amlogic; setenv avb2 0;
storeboot=boot_cooling; run check_button; run get_boot_slot; run get_kernel_slot; run storeargs; get_valid_slot; setenv bootargs ${bootargs} androidboot.slot_suffix=${active_slot}; consume_boot_try; if run storeload; then booti 0x1080000 - 0x1000000; fi; run update;
storeload=fatload mmc 1:1 0x1000000 superbird.dtb; fatload mmc 1:${kernel_part} 0x1080000 Image;
